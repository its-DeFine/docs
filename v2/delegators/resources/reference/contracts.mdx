import { contractAddresses } from '/snippets/data/contract-addresses/contractAddressesData.jsx'

export const arbitrumContracts = [
  'BondingManager',
  'Minter',
  'RoundsManager',
  'BondingVotes',
  'LivepeerGovernor',
  'L2LPTGateway',
  'LivepeerToken'
]
  .map((name) => contractAddresses.arbitrumOne.inventory.find((entry) => entry.name === name && entry.lifecycle === 'active'))
  .filter(Boolean)

export const ethereumContracts = [
  'L1Escrow',
  'LivepeerToken'
]
  .map((name) => contractAddresses.ethereumMainnet.inventory.find((entry) => entry.name === name && entry.lifecycle === 'active'))
  .filter(Boolean)

<Warning>
  This page renders from the repo's generated contract-address dataset. If you need to verify a value before signing a transaction, open the linked block explorer entry and check the tagged contract there as well.
</Warning>
